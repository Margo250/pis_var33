package announcement;

service AnnouncementService {
    // Команды (Command)
    rpc CreateAnnouncement(CreateAnnouncementRequest) returns (CreateAnnouncementResponse);
    rpc PublishAnnouncement(PublishAnnouncementRequest) returns (PublishAnnouncementResponse);
    rpc ScheduleAnnouncement(ScheduleAnnouncementRequest) returns (ScheduleAnnouncementResponse);

### 3. gRPC Server (Announcement Service)

**Файл: `src/infrastructure/grpc/server/announcement_grpc_server.py`**
```python
import grpc
from concurrent import futures
from typing import List, Dict, Set
import asyncio
from datetime import datetime

from ..generated import announcement_service_pb2 as pb2
from ..generated import announcement_service_pb2_grpc as pb2_grpc

from application.service.announcement_service_impl import AnnouncementServiceImpl
from application.command.create_announcement_command import CreateAnnouncementCommand
from application.command.publish_announcement_command import PublishAnnouncementCommand
from application.command.schedule_announcement_command import ScheduleAnnouncementCommand
from application.command.add_attachment_command import AddAttachmentCommand
from application.command.remove_attachment_command import RemoveAttachmentCommand
from application.command.archive_announcement_command import ArchiveAnnouncementCommand
from application.query.get_announcement_by_id_query import GetAnnouncementByIdQuery
from application.query.list_announcements_by_group_query import ListAnnouncementsByGroupQuery


class AnnouncementGrpcService(pb2_grpc.AnnouncementServiceServicer):
    """gRPC сервер для Announcement Service"""
    
    def __init__(self, app_service: AnnouncementServiceImpl):
        self._app_service = app_service
        self._subscribers: Dict[str, Set[grpc.ServicerContext]] = {}  # group_id -> contexts
    
    def CreateAnnouncement(self, request, context):
        """Создать объявление"""
        try:
            command = CreateAnnouncementCommand(
                group_id=request.group_id,
                author_id=request.author_id,
                title=request.title,
                content=request.content,
                attachment_ids=list(request.attachment_ids) if request.attachment_ids else None
            )
            announcement_id = self._app_service.create_announcement(command)
            return pb2.CreateAnnouncementResponse(announcement_id=announcement_id)
        except ValueError as e:
            context.set_code(grpc.StatusCode.INVALID_ARGUMENT)
            context.set_details(str(e))
            return pb2.CreateAnnouncementResponse()
        except Exception as e:
            context.set_code(grpc.StatusCode.INTERNAL)
            context.set_details(str(e))
            return pb2.CreateAnnouncementResponse()
    
    def PublishAnnouncement(self, request, context):
        """Опубликовать объявление"""
        try:
            command = PublishAnnouncementCommand(announcement_id=request.announcement_id)
            self._app_service.publish_announcement(command)
            
            # Уведомить подписчиков группы
            announcement = self._app_service.get_announcement_by_id(
                GetAnnouncementByIdQuery(request.announcement_id)
            )
            self._notify_subscribers(announcement.group_id, "published", announcement)
            
            return pb2.PublishAnnouncementResponse(success=True, message="Published")
        except ValueError as e:
            context.set_code(grpc.StatusCode.NOT_FOUND)
            context.set_details(str(e))
            return pb2.PublishAnnouncementResponse(success=False, message=str(e))
    
    def ScheduleAnnouncement(self, request, context):
        """Запланировать публикацию"""
        try:
            command = ScheduleAnnouncementCommand(
                announcement_id=request.announcement_id,
                scheduled_for=datetime.fromisoformat(request.scheduled_for)
            )
            self._app_service.schedule_announcement(command)
            return pb2.ScheduleAnnouncementResponse(success=True)
        except ValueError as e:
            context.set_code(grpc.StatusCode.INVALID_ARGUMENT)
            context.set_details(str(e))
            return pb2.ScheduleAnnouncementResponse(success=False)
    
    def AddAttachment(self, request, context):
        """Добавить вложение"""
        try:
            command = AddAttachmentCommand(
                announcement_id=request.announcement_id,
                file_id=request.file_id,
                filename=request.filename,
                file_size=request.file_size,
                mime_type=request.mime_type,
                storage_path=request.storage_path
            )
            self._app_service.add_attachment(command)
            return pb2.AddAttachmentResponse(success=True)
        except ValueError as e:
            context.set_code(grpc.StatusCode.INVALID_ARGUMENT)
            context.set_details(str(e))
            return pb2.AddAttachmentResponse(success=False)
    
    def RemoveAttachment(self, request, context):
        """Удалить вложение"""
        try:
            command = RemoveAttachmentCommand(
                announcement_id=request.announcement_id,
                file_id=request.file_id
            )
            self._app_service.remove_attachment(command)
            return pb2.RemoveAttachmentResponse(success=True)
        except ValueError as e:
            context.set_code(grpc.StatusCode.NOT_FOUND)
            context.set_details(str(e))
            return pb2.RemoveAttachmentResponse(success=False)
    
    def ArchiveAnnouncement(self, request, context):
        """Архивировать объявление"""
        try:
            command = ArchiveAnnouncementCommand(announcement_id=request.announcement_id)
            self._app_service.archive_announcement(command)
            return pb2.ArchiveAnnouncementResponse(success=True)
        except ValueError as e:
            context.set_code(grpc.StatusCode.NOT_FOUND)
            context.set_details(str(e))
            return pb2.ArchiveAnnouncementResponse(success=False)
    
    def GetAnnouncement(self, request, context):
        """Получить объявление"""
        try:
            query = GetAnnouncementByIdQuery(announcement_id=request.announcement_id)
            dto = self._app_service.get_announcement_by_id(query)
            
            return pb2.GetAnnouncementResponse(
                id=dto.id,
                group_id=dto.group_id,
                author_id=dto.author_id,
                title=dto.title,
                content=dto.content,
                status=dto.status,
                attachments=[
                    pb2.AttachmentDto(
                        file_id=a.file_id,
                        filename=a.filename,
                        file_size=a.file_size,
                        mime_type=a.mime_type
                    ) for a in dto.attachments
                ],
                created_at=dto.created_at.isoformat(),
                published_at=dto.published_at.isoformat() if dto.published_at else "",
                scheduled_for=dto.scheduled_for.isoformat() if dto.scheduled_for else "",
                updated_at=dto.updated_at.isoformat()
            )
        except ValueError as e:
            context.set_code(grpc.StatusCode.NOT_FOUND)
            context.set_details(str(e))
            return pb2.GetAnnouncementResponse()
    
    def ListAnnouncementsByGroup(self, request, context):
        """Список объявлений по группе"""
        query = ListAnnouncementsByGroupQuery(
            group_id=request.group_id,
            limit=request.limit if request.limit else 50,
            offset=request.offset if request.offset else 0
        )
        dtos = self._app_service.list_announcements_by_group(query)
        
        announcements = []
        for dto in dtos:
            announcements.append(pb2.GetAnnouncementResponse(
                id=dto.id,
                group_id=dto.group_id,
                author_id=dto.author_id,
                title=dto.title,
                content=dto.content,
                status=dto.status,
                attachments=[
                    pb2.AttachmentDto(
                        file_id=a.file_id,
                        filename=a.filename,
                        file_size=a.file_size,
                        mime_type=a.mime_type
                    ) for a in dto.attachments
                ],
                created_at=dto.created_at.isoformat(),
                published_at=dto.published_at.isoformat() if dto.published_at else "",
                scheduled_for=dto.scheduled_for.isoformat() if dto.scheduled_for else "",
                updated_at=dto.updated_at.isoformat()
            ))
        
        return pb2.ListAnnouncementsResponse(announcements=announcements, total=len(announcements))
    
    def SubscribeToNewAnnouncements(self, request, context):
        """Server-side streaming: подписка на новые объявления"""
        group_id = request.group_id
        client_id = request.client_id
        
        # Регистрация подписчика
        if group_id not in self._subscribers:
            self._subscribers[group_id] = set()
        self._subscribers[group_id].add(context)
        
        try:
            # Держим соединение открытым
            while True:
                # Проверяем, не закрыто ли соединение
                if context.is_active():
                    asyncio.sleep(1)
                else:
                    break
        finally:
            # Отписка при закрытии
            self._subscribers[group_id].discard(context)
            if not self._subscribers[group_id]:
                del self._subscribers[group_id]
    
    def _notify_subscribers(self, group_id: str, event_type: str, announcement):
        """Уведомить всех подписчиков группы"""
        if group_id not in self._subscribers:
            return
        
        event = pb2.AnnouncementStreamEvent(
            event_type=event_type,
            announcement=pb2.GetAnnouncementResponse(
                id=announcement.id,
                group_id=announcement.group_id,
                author_id=announcement.author_id,
                title=announcement.title,
                content=announcement.content,
                status=announcement.status,
                created_at=announcement.created_at.isoformat(),
                updated_at=announcement.updated_at.isoformat()
            ),
            occurred_at=datetime.now().isoformat()
        )
        
        dead_contexts = []
        for context in self._subscribers[group_id]:
            try:
                context.write(event)
            except Exception:
                dead_contexts.append(context)
        
        # Удалить мёртвые соединения
        for dc in dead_contexts:
            self._subscribers[group_id].discard(dc)


def serve_grpc(announcement_service, port: int = 50051):
    """Запуск gRPC сервера"""
    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
    pb2_grpc.add_AnnouncementServiceServicer_to_server(
        AnnouncementGrpcService(announcement_service), server
    )
    server.add_insecure_port(f'[::]:{port}')
    server.start()
    print(f"gRPC server started on port {port}")
    server.wait_for_termination()
```

import grpc
from typing import List, Optional
from ..generated import group_service_pb2 as pb2
from ..generated import group_service_pb2_grpc as pb2_grpc


class GroupGrpcClient:
    """gRPC клиент для Group Service"""
    
    def __init__(self, channel_address: str = "localhost:50052"):
        self._channel = grpc.insecure_channel(channel_address)
        self._stub = pb2_grpc.GroupServiceStub(self._channel)
    
    def get_group(self, group_id: str) -> Optional[dict]:
        """Получить информацию о группе"""
        try:
            request = pb2.GetGroupRequest(group_id=group_id)
            response = self._stub.GetGroup(request)
            
            if response.id:
                return {
                    "id": response.id,
                    "name": response.name,
                    "owner_id": response.owner_id,
                    "member_count": response.member_count,
                    "admin_ids": list(response.admin_ids),
                    "created_at": response.created_at
                }
            return None
        except grpc.RpcError as e:
            if e.code() == grpc.StatusCode.NOT_FOUND:
                return None
            raise
    
    def get_group_members(self, group_id: str, limit: int = 100) -> List[dict]:
        """Получить список участников группы"""
        try:
            request = pb2.GetGroupMembersRequest(
                group_id=group_id,
                limit=limit,
                offset=0
            )
            response = self._stub.GetGroupMembers(request)
            
            return [
                {
                    "user_id": m.user_id,
                    "email": m.email,
                    "name": m.name,
                    "role": m.role,
                    "joined_at": m.joined_at
                }
                for m in response.members
            ]
        except grpc.RpcError:
            return []
    
    def check_member_exists(self, group_id: str, user_id: str) -> tuple:
        """Проверить, состоит ли пользователь в группе"""
        try:
            request = pb2.CheckMemberRequest(
                group_id=group_id,
                user_id=user_id
            )
            response = self._stub.CheckMember(request)
            return (response.is_member, response.is_admin)
        except grpc.RpcError:
            return (False, False)
    
    def close(self):
        """Закрыть соединение"""
        self._channel.close()

# tests/grpc/test_streaming.py

import grpc
import sys
from src.infrastructure.grpc.generated import announcement_service_pb2 as pb2
from src.infrastructure.grpc.generated import announcement_service_pb2_grpc as pb2_grpc


def subscribe_to_announcements(group_id: str, client_id: str):
    """Подписка на новые объявления в группе"""
    
    channel = grpc.insecure_channel('localhost:50051')
    stub = pb2_grpc.AnnouncementServiceStub(channel)
    
    request = pb2.SubscribeRequest(
        group_id=group_id,
        client_id=client_id
    )
    
    print(f"Подключение к группе {group_id}...")
    
    try:
        for event in stub.SubscribeToNewAnnouncements(request):
            print(f"\n[{event.occurred_at}] Событие: {event.event_type}")
            announcement = event.announcement
            print(f"  Заголовок: {announcement.title}")
            print(f"  Статус: {announcement.status}")
            
            if event.event_type == "published":
                print(f"  🎉 Новое объявление в группе!")
    
    except grpc.RpcError as e:
        print(f"Ошибка: {e}")
    
    except KeyboardInterrupt:
        print("\nОтключение...")
        channel.close()


if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: python client.py <group_id> <client_id>")
        sys.exit(1)
    
    subscribe_to_announcements(sys.argv[1], sys.argv[2])
