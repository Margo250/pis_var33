| DELETE | `/api/announcements/attachments` | Удалить вложение |
| POST | `/api/announcements/archive` | Архивировать |
| PUT | `/api/announcements/content` | Обновить содержимое |
| GET | `/api/announcements/{id}` | Получить объявление |
| GET | `/api/announcements/` | Список по группе |

**Зависимости от других сервисов:**
- При создании объявления — проверяет существование группы через Group Service (с кэшированием)
- При добавлении файла — запрашивает metadata у File Service

**Пример кода (REST клиент к Group Service):**

```python
# infrastructure/clients/group_service_client.py

import httpx
from typing import Optional
from tenacity import retry, stop_after_attempt, wait_exponential


class GroupServiceClient:
    """HTTP клиент для взаимодействия с Group Service"""
    
    def __init__(self, base_url: str = "http://group-service:8001"):
        self._base_url = base_url
        self._cache = {}  # Простой in-memory кэш
    
    @retry(stop=stop_after_attempt(3), wait=wait_exponential(multiplier=1, min=1, max=10))
    async def get_group(self, group_id: str) -> Optional[dict]:
        """Получить информацию о группе"""
        # Проверка кэша
        if group_id in self._cache:
            return self._cache[group_id]
        
        async with httpx.AsyncClient() as client:
            response = await client.get(f"{self._base_url}/api/groups/{group_id}")
            
            if response.status_code == 200:
                data = response.json()
                self._cache[group_id] = data
                return data
            elif response.status_code == 404:
                return None
            else:
                response.raise_for_status()
    
    async def get_group_members(self, group_id: str) -> list:
        """Получить список участников группы"""
        async with httpx.AsyncClient() as client:
            response = await client.get(f"{self._base_url}/api/groups/{group_id}/members")
            if response.status_code == 200:
                return response.json()
            return []
    
    def clear_cache(self, group_id: str = None):
        """Очистить кэш"""
        if group_id:
            self._cache.pop(group_id, None)
        else:
            self._cache.clear()
```

### 3. Notification Service

**Bounded Context**: Асинхронная отправка уведомлений

**Подписывается на события**:
- AnnouncementPublishedEvent → уведомить всех участников группы
- GroupAssignedEvent → уведомить группу о назначении
- AttachmentAddedEvent → уведомить автора

**Код подписчика**:
```python
# application/subscribers/notification_subscriber.py

import asyncio
from typing import Dict, Callable
import aio_pika


class NotificationSubscriber:
    """Подписчик на события для отправки уведомлений"""
    
    def __init__(self, rabbitmq_url: str, email_service, push_service):
        self._rabbitmq_url = rabbitmq_url
        self._email_service = email_service
        self._push_service = push_service
        self._handlers: Dict[str, Callable] = {
            "announcement.published": self._handle_announcement_published,
            "announcement.attachment.added": self._handle_attachment_added,
        }
    
    async def start(self):
        """Запуск слушателя очереди"""
        connection = await aio_pika.connect_robust(self._rabbitmq_url)
        
        async with connection:
            channel = await connection.channel()
            queue = await channel.declare_queue("notification_queue", durable=True)
            
            async with queue.iterator() as queue_iter:
                async for message in queue_iter:
                    async with message.process():
                        body = json.loads(message.body)
                        event_type = body.get("event_type")
                        
                        handler = self._handlers.get(event_type)
                        if handler:
                            await handler(body)
    
    async def _handle_announcement_published(self, event: dict):
        """Отправить уведомления всем участникам группы"""
        announcement_id = event["announcement_id"]
        group_id = event["group_id"]
        
        # Получить список участников группы через Group Service
        members = await self._group_client.get_group_members(group_id)
        
        # Отправить уведомления (email + push)
        tasks = []
        for member in members:
            tasks.append(self._email_service.send(
                to=member["email"],
                subject=f"Новое объявление в группе",
                body=f"Опубликовано новое объявление #{announcement_id[:8]}"
            ))
        
        await asyncio.gather(*tasks)
```

### 4. Event Bus (RabbitMQ)

**Конфигурация обменников и очередей**:
```python
# infrastructure/event_bus/rabbitmq_event_bus.py

import json
import pika
from typing import List


class RabbitMQEventBus:
    """Реализация Event Bus через RabbitMQ"""
    
    EXCHANGE_NAME = "announcement.events"
    
    def __init__(self, host: str = "localhost"):
        self._host = host
        self._connection = None
        self._channel = None
    
    def connect(self):
        """Установить соединение"""
        self._connection = pika.BlockingConnection(
            pika.ConnectionParameters(host=self._host)
        )
        self._channel = self._connection.channel()
        
        # Создать topic exchange
        self._channel.exchange_declare(
            exchange=self.EXCHANGE_NAME,
            exchange_type="topic",
            durable=True
        )
    
    def publish(self, event_name: str, payload: dict, routing_key: str = None):
        """Опубликовать событие"""
        if not self._channel:
            self.connect()
        
        routing_key = routing_key or event_name
        
        message = {
            "event_type": event_name,
            "payload": payload,
            "timestamp": datetime.now().isoformat()
        }
        
        self._channel.basic_publish(
            exchange=self.EXCHANGE_NAME,
            routing_key=routing_key,
            body=json.dumps(message),
            properties=pika.BasicProperties(delivery_mode=2)  # persistent
        )
    
    def subscribe(self, routing_key: str, queue_name: str, callback):
        """Подписаться на события"""
        if not self._channel:
            self.connect()
        
        # Создать очередь
        self._channel.queue_declare(queue=queue_name, durable=True)
        
        # Привязать к exchange
        self._channel.queue_bind(
            exchange=self.EXCHANGE_NAME,
            queue=queue_name,
            routing_key=routing_key
        )
        
        # Начать потребление
        self._channel.basic_consume(
            queue=queue_name,
            on_message_callback=callback,
            auto_ack=False
        )
    
    def start_consuming(self):
        """Запустить обработку сообщений"""
        if self._channel:
            self._channel.start_consuming()
```

**Схема обменников**:

```
                    ┌─────────────────────────────────────────┐
                    │         announcement.events             │
                    │           (Topic Exchange)              │
                    └─────────────────────┬───────────────────┘
                                          │
            ┌─────────────────────────────┼─────────────────────────────┐
            │                             │                             │
            ▼                             ▼                             ▼
    ┌───────────────┐             ┌───────────────┐             ┌───────────────┐
    │ notification  │             │   analytics   │             │   logging     │
    │    queue      │             │    queue      │             │    queue      │
    │ routing: *.   │             │ routing: *.   │             │ routing: #    │
    │ published     │             │ created       │             │               │
    └───────┬───────┘             └───────┬───────┘             └───────┬───────┘
            │                             │                             │
            ▼                             ▼                             ▼
    ┌───────────────┐             ┌───────────────┐             ┌───────────────┐
    │ Notification  │             │   Analytics   │             │    Logger     │
```

### 6. Circuit Breaker

**Реализация Circuit Breaker для вызовов между сервисами**:
```python
# infrastructure/circuit_breaker/circuit_breaker.py

from enum import Enum
from datetime import datetime, timedelta
from typing import Callable, TypeVar
import logging

T = TypeVar('T')


class CircuitState(Enum):
    CLOSED = "closed"      # Работает нормально
    OPEN = "open"          # Разрыв цепи (сервис недоступен)
    HALF_OPEN = "half_open"  # Проверка восстановления


class CircuitBreaker:
    """Реализация Circuit Breaker паттерна"""
    
    def __init__(self, name: str, failure_threshold: int = 5, 
                 timeout_seconds: int = 30, half_open_max_calls: int = 3):
        self.name = name
        self._failure_threshold = failure_threshold
        self._timeout_seconds = timeout_seconds
        self._half_open_max_calls = half_open_max_calls
        
        self._state = CircuitState.CLOSED
        self._failure_count = 0
        self._last_failure_time = None
        self._half_open_calls = 0
        self._logger = logging.getLogger(f"circuit_breaker.{name}")
    
    @property
    def state(self) -> CircuitState:
        """Получить текущее состояние с авто-восстановлением"""
        if self._state == CircuitState.OPEN:
            if datetime.now() - self._last_failure_time > timedelta(seconds=self._timeout_seconds):
                self._transition_to_half_open()
        return self._state
    
    def call(self, func: Callable[..., T], *args, **kwargs) -> T:
        """Вызвать функцию с защитой Circuit Breaker"""
        if self.state == CircuitState.OPEN:
            raise CircuitBreakerOpenException(f"Circuit {self.name} is OPEN")
        
        try:
            result = func(*args, **kwargs)
            self._on_success()
            return result
        except Exception as e:
            self._on_failure(e)
            raise
    
    def _on_success(self):
        """Обработка успешного вызова"""
        if self._state == CircuitState.HALF_OPEN:
            self._half_open_calls += 1
            if self._half_open_calls >= self._half_open_max_calls:
                self._transition_to_closed()
        else:
            self._failure_count = 0
            self._last_failure_time = None
    
    def _on_failure(self, error: Exception):
        """Обработка неудачного вызова"""
        self._failure_count += 1
        self._last_failure_time = datetime.now()
        
        if self._state == CircuitState.HALF_OPEN:
            self._transition_to_open()
        elif self._failure_count >= self._failure_threshold:
            self._transition_to_open()
        
        self._logger.warning(f"Circuit {self.name} recorded failure #{self._failure_count}: {error}")
    
    def _transition_to_closed(self):
        """Переход в CLOSED состояние"""
        self._state = CircuitState.CLOSED
        self._failure_count = 0
        self._half_open_calls = 0
        self._logger.info(f"Circuit {self.name} transitioned to CLOSED")
    
    def _transition_to_open(self):
        """Переход в OPEN состояние"""
        self._state = CircuitState.OPEN
        self._logger.warning(f"Circuit {self.name} transitioned to OPEN after {self._failure_count} failures")
    
    def _transition_to_half_open(self):
        """Переход в HALF_OPEN состояние"""
        self._state = CircuitState.HALF_OPEN
        self._half_open_calls = 0
        self._logger.info(f"Circuit {self.name} transitioned to HALF_OPEN")


class CircuitBreakerOpenException(Exception):
    """Исключение при разомкнутом Circuit Breaker"""
    pass


# Пример использования
circuit_breaker = CircuitBreaker("group_service", failure_threshold=3, timeout_seconds=30)

def get_group_with_circuit(group_id: str):
    return circuit_breaker.call(group_client.get_group, group_id)
```
